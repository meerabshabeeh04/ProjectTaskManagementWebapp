CREATE TABLE USER_(
      USER_ID INT AUTO_INCREMENT PRIMARY KEY,
      FIRST_NAME VARCHAR(250),
      LAST_NAME VARCHAR(250),
      EMAIL VARCHAR(250),
      PASSWORD_ VARCHAR(250),
      PH_NO BIGINT,
      GENDER VARCHAR(250)
);


CREATE TABLE PROJECT(
      PROJECT_ID INT AUTO_INCREMENT PRIMARY KEY,
      OWNER_ID INT,
      TITLE VARCHAR(250),
      DESCRIPTION VARCHAR(5000),
      CONSTRAINT FOREIGN KEY(OWNER_ID) REFERENCES USER_(USER_ID)
);


CREATE TABLE TASKS(
      TASK_NO INT AUTO_INCREMENT PRIMARY KEY,
      PROJECT_ID INT,
      DESCRIPTION VARCHAR(5000),
      DEADLINE DATE,
      STATUS VARCHAR(250),
      CONSTRAINT FOREIGN KEY(PROJECT_ID) REFERENCES PROJECT(PROJECT_ID)       
);


CREATE TABLE USER_TASK(
       TASK_NO INT PRIMARY KEY,
       USER_ID INT,
       CONSTRAINT FOREIGN KEY(TASK_NO) REFERENCES TASKS(TASK_NO),
       CONSTRAINT FOREIGN KEY(USER_ID) REFERENCES USER_(USER_ID)
);
